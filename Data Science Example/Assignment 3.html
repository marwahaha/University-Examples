def compute_upgma_tree(matrix):

    import itertools as it

    n = len(matrix)
    nodes = [Node(str(i + 1)) for i in range(n)]
    for node in nodes:
        node.set_height(0)
    matrix = {nodes[i]: {nodes[j]: matrix[i][j] for j in range(n)} for i in range(n)}
    while len(matrix) > 1:
        a, b = min(it.combinations(matrix.keys(), 2), key=lambda xy: matrix[xy[0]][xy[1]])
        u = Node()
        u.add_child(a)
        u.add_child(b)
        u.set_height(matrix[a][b] / 2)
        uc = {c: (a.get_leaf_count() * matrix[a][c] + b.get_leaf_count() * matrix[b][c]) / (a.get_leaf_count() + b.get_leaf_count()) for c in matrix.keys() - set((a, b))}
        del matrix[a]
        del matrix[b]
        for k, v in matrix.items():
            del v[a]
            del v[b]
            v[u] = uc[k]
        matrix[u] = uc
    return Tree(u)

        self.height = height

        return not self.children

        return self.sequence

        self.sequence = sequence

        return self.label

        self.label = label

    def get_leaves(self):
        leaf_list = []

        if (self.is_leaf()):
            leaf_list.append(self)
        else:
            for child in self.children:
                leaf_list.extend(child.get_leaves())

        return leaf_list

        if self.is_leaf():
            return 1
        else:
            return sum(map(Node.get_leaf_count, self.children))

        if (not self.is_root()):
            branch_length = self.parent.height - self.height
        else:
            branch_length = 0.0

        sb += ":" + str(branch_length)

        return sb

def jukes_cantor(tree, sequence_length, mutation_rate):
    # Set the root
    current_node = tree.get_root()
    current_node.set_sequence(rand_seq(sequence_length))
        
    # If children exist, visit them
    children = current_node.get_children()
    if (children):

# Visits every child of a given node, mutates the given node
def traverse(node, sequence_length, mutation_rate):    
    # Mutate the parent sequence and assign to this node
    parent_seq = node.get_parent().get_sequence()
    time = node.get_parent().get_height() - node.get_height()
    node.set_sequence(copy.deepcopy(mutate(parent_seq, time, mutation_rate)))
    
    # Recursive call on the children
    if not(node.is_leaf()):
        children = node.get_children()
        if (children):
            for child in children:
                traverse(child, sequence_length, mutation_rate)

    
# produces a random genome or length L
# and returns that random genome
def rand_seq(L):
    seq = []
    for i in range(L):
        seq += random.choice("ACGT")
    return seq

# Given a sequence, time and mutation rate,
# mutates that sequence based on the mutation rate
# and time. Returns the mutated sequence
def mutate(X,t,mu):
    L = len(X)
    # the number of mutations is Poisson with total rate L*mu*t
    numMutation = np.random.poisson(L*mu*t)
    # for each mutation, choose a site to mutate and mutate it
    for i in range(numMutation):
        # choose a site
        site = math.floor(random.random()*L)
        # mutate that site
        X[site] = random.choice("ACGT")
    return X

def create_distance_matrix(sequence_list):
    numpy_matrix = []
    for i in range(len(sequence_list)):
        numpy_row = []

def p_T_Ne(n, Ne, tk_list):
    total = 0
    i = 0
    for k in range(2, n + 1):
        total += -(math.log(Ne)) - ( (k * (k - 1) * tk_list[i]) / (2 * Ne) )
        i += 1
    return total

def accept(pre_rate, post_rate):
    try: 
        accept_rate = 1 - min(1, math.exp(pre_rate - post_rate))
    except:
        accept_rate = 0
    rand = np.random.uniform(0,1)
    if (accept_rate > rand):
        return True
    else:
        return False

Observations¶

  • The distribution of my sampled values of Ne (orange) roughly folows a log normal distribution (blue) indicating that my sampled Ne is a log normal distribution
  • The peak of the lognormal distribution corresponds with the real value of Ne.
  • Because I believe that the population is log normally distributed, I can say that the distribution is unimodal. If I can safely make this assumption, then I can reduce the step size so my kernel proposes smaller steps. This is fine for unimodal data because getting stuck at a peak is not a problem.
    • Smaller steps will increase the number of steps required to find a stable estimate for Ne
    • Smaller steps will more assuredly find a local maximum
  • My acceptance rate seems ~25% which is roughly ideal.
  • If the acceptance rate is too low, it could indicate the step size is too large relative to the population size
  • Having an idea of a starting Ne could help reduce the burn in period of the graph
  • Modifying the acceptance rate would modify how often the kernel explores populations but would reduce the speed of convergence
  • When the mean flattens, we can see our algorithm has somewhat converged on a true value of Ne. If it is still moving I am inclined to believe the algorithm needs to be run longer.
